from typing import Any, Literal

from pydantic import BaseModel, Field

INVALID_REQUEST = -32600
METHOD_NOT_FOUND = -32601
INVALID_PARAMS = -32602
INTERNAL_ERROR = -32603
PARSE_ERROR = -32700


class JsonRpcRequest(BaseModel):
    jsonrpc: Literal["2.0"] = "2.0"
    id: str
    method: str = Field(min_length=1)
    params: dict[str, Any] = Field(default_factory=dict)


class JsonRpcSuccess(BaseModel):
    jsonrpc: Literal["2.0"] = "2.0"
    id: str
    result: dict[str, Any]


class JsonRpcErrorBody(BaseModel):
    code: int
    message: str
    data: dict[str, Any] | None = None


class JsonRpcError(BaseModel):
    jsonrpc: Literal["2.0"] = "2.0"
    id: str | None
    error: JsonRpcErrorBody


# 生成标准 JSON-RPC 错误响应
def make_error(
    *,
    id: str | None,
    code: int,
    message: str,
    data: dict[str, Any] | None = None,
) -> JsonRpcError:
    return JsonRpcError(id=id, error=JsonRpcErrorBody(code=code, message=message, data=data))
```

import os
from collections.abc import Mapping
from dataclasses import dataclass


@dataclass(frozen=True)
class CoreConfig:
    host: str = "127.0.0.1"
    port: int = 7437
    log_level: str = "INFO"


# 从环境变量加载 core 配置
def load_config(env: Mapping[str, str] | None = None) -> CoreConfig:
    source = os.environ if env is None else env
    raw_port = source.get("GT_PORT", "7437")
    try:
        port = int(raw_port)
    except ValueError as exc:
        raise ValueError("GT_PORT must be an integer") from exc

    return CoreConfig(
        host=source.get("GT_HOST", "127.0.0.1"),
        port=port,
        log_level=source.get("GT_LOG_LEVEL", "INFO"),
    )
```

from gt_claude.core.transport.socket_server import SocketServer


# 功能：验证 socket server 能把 JSON-RPC 请求分发给注册 handler
# 设计：启动真实 asyncio TCP server，用 NDJSON 往返覆盖传输层边界
def test_socket_server_dispatches_registered_method(event_loop: asyncio.AbstractEventLoop) -> None:
    async def scenario() -> None:
        server = SocketServer(host="127.0.0.1", port=0)

        async def handle(params: dict[str, object]) -> dict[str, object]:
            return {"echo": params["value"]}

        server.register("test.echo", handle)
        await server.start()
        assert server.bound_port is not None
        try:
            reader, writer = await asyncio.open_connection("127.0.0.1", server.bound_port)
            writer.write(b'{"jsonrpc":"2.0","id":"u-1","method":"test.echo","params":{"value":"ok"}}\n')
            await writer.drain()
            line = await reader.readline()
            writer.close()
            await writer.wait_closed()
        finally:
            await server.stop()

        payload = json.loads(line.decode())
        assert payload == {"jsonrpc": "2.0", "id": "u-1", "result": {"echo": "ok"}}

    event_loop.run_until_complete(scenario())

```python
import asyncio
import json
import logging
from collections.abc import Awaitable, Callable
from typing import Any

from pydantic import ValidationError

from gt_claude.core.bus.envelope import (
    INTERNAL_ERROR,
    INVALID_REQUEST,
    METHOD_NOT_FOUND,
    JsonRpcRequest,
    JsonRpcSuccess,
    make_error,
)

Handler = Callable[[dict[str, Any]], Awaitable[dict[str, Any]]]

logger = logging.getLogger(__name__)


class SocketServer:
    def __init__(self, *, host: str, port: int) -> None:
        self.host = host
        self.port = port
        self.bound_port: int | None = None
        self._server: asyncio.Server | None = None
        self._handlers: dict[str, Handler] = {}

    # 注册 JSON-RPC method 对应的异步处理函数
    def register(self, method: str, handler: Handler) -> None:
        self._handlers[method] = handler

    # 启动 TCP NDJSON server
    async def start(self) -> None:
        self._server = await asyncio.start_server(self._handle_client, self.host, self.port)
        socket = self._server.sockets[0]
        self.bound_port = int(socket.getsockname()[1])
        logger.info("gt-core listening on %s:%s", self.host, self.bound_port)

    # 停止 TCP server 并等待关闭完成
    async def stop(self) -> None:
        if self._server is None:
            return
        self._server.close()
        await self._server.wait_closed()
        self._server = None

    # 处理单个客户端连接中的 NDJSON 请求
    async def _handle_client(self, reader: asyncio.StreamReader, writer: asyncio.StreamWriter) -> None:
        try:
            while line := await reader.readline():
                response = await self._dispatch_line(line)
                writer.write(response.model_dump_json(exclude_none=True).encode() + b"\n")
                await writer.drain()
        finally:
            writer.close()
            await writer.wait_closed()

    # 分发一行 JSON-RPC 请求并返回响应 envelope
    async def _dispatch_line(self, line: bytes) -> JsonRpcSuccess | object:
        try:
            raw = json.loads(line.decode())
            req = JsonRpcRequest.model_validate(raw)
        except (json.JSONDecodeError, UnicodeDecodeError, ValidationError) as exc:
            return make_error(id=None, code=INVALID_REQUEST, message=str(exc))

        handler = self._handlers.get(req.method)
        if handler is None:
            return make_error(id=req.id, code=METHOD_NOT_FOUND, message=f"method not found: {req.method}")

        try:
            result = await handler(req.params)
        except Exception as exc:
            logger.exception("handler failed for method %s", req.method)
            return make_error(id=req.id, code=INTERNAL_ERROR, message=str(exc))

        return JsonRpcSuccess(id=req.id, result=result)
```

```python
import asyncio
import signal
import time
from datetime import UTC, datetime
from typing import Any

from gt_claude import __version__
from gt_claude.core.bus.commands import PingCommand, PongResult
from gt_claude.core.config import load_config
from gt_claude.core.logging_setup import setup_logging
from gt_claude.core.transport.socket_server import SocketServer


class CoreApp:
    def __init__(self) -> None:
        self.config = load_config()
        self.started_at = time.perf_counter()
        self.server = SocketServer(host=self.config.host, port=self.config.port)

    # 启动 core app 并等待系统信号退出
    async def run(self) -> None:
        setup_logging(self.config.log_level)
        self.server.register("core.ping", self._handle_ping)
        await self.server.start()
        stop_event = asyncio.Event()
        loop = asyncio.get_running_loop()
        for sig in (signal.SIGINT, signal.SIGTERM):
            loop.add_signal_handler(sig, stop_event.set)
        try:
            await stop_event.wait()
        finally:
            await self.server.stop()

    # 处理 core.ping 命令
    async def _handle_ping(self, params: dict[str, Any]) -> dict[str, Any]:
        PingCommand.model_validate(params)
        uptime_ms = int((time.perf_counter() - self.started_at) * 1000)
        result = PongResult(
            server_version=__version__,
            uptime_ms=uptime_ms,
            received_at=datetime.now(UTC).isoformat(),
        )
        return result.model_dump()


# 运行 gt-core daemon
async def async_run() -> None:
    await CoreApp().run()


# 同步入口，供 console script 调用
def run() -> None:
    asyncio.run(async_run())
```

import argparse
from pathlib import Path
from typing import Any

from pydantic import BaseModel

from gt_claude.core.bus.commands import PingCommand, PongResult
from gt_claude.core.bus.events import CoreStartedEvent

ROOT = Path(__file__).resolve().parents[1]
OUT = ROOT / "WIRE_PROTOCOL.md"


# 将 Pydantic schema 的 properties 渲染成 Markdown 表格
def render_fields(model: type[BaseModel]) -> str:
    schema: dict[str, Any] = model.model_json_schema()
    properties: dict[str, Any] = schema.get("properties", {})
    required = set(schema.get("required", []))
    lines = ["| Field | Type | Required |", "|---|---|---|"]
    for name, prop in properties.items():
        type_name = prop.get("type", "const" if "const" in prop else "object")
        lines.append(f"| `{name}` | `{type_name}` | {'yes' if name in required else 'no'} |")
    return "\n".join(lines)


# 生成完整 wire protocol Markdown 内容

def generate() -> str:
    return f"""# Wire Protocol

> Generated by `scripts/gen_protocol_doc.py`. Do not edit manually.

## Transport

- TCP loopback `127.0.0.1:7437` by default.
- Override host/port with `GT_HOST` and `GT_PORT`.
- Each message is one `\\n`-terminated JSON line.
- Commands use JSON-RPC 2.0 from client to server.

## Commands

### PingCommand

{render_fields(PingCommand)}

```json
{{
  "jsonrpc": "2.0",
  "id": "u-1",
  "method": "core.ping",
  "params": {{
    "client": "gt/0.1.0"
  }}
}}
```

### PongResult

{render_fields(PongResult)}

```json
{{
  "jsonrpc": "2.0",
  "id": "u-1",
  "result": {{
    "server_version": "0.1.0",
    "uptime_ms": 12,
    "received_at": "2026-06-25T00:00:00+00:00"
  }}
}}
```

## Events

### CoreStartedEvent

{render_fields(CoreStartedEvent)}
"""


# 写入或校验 WIRE_PROTOCOL.md

def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--check", action="store_true", help="verify WIRE_PROTOCOL.md is up to date")
    args = parser.parse_args()
    content = generate()
    if args.check:
        current = OUT.read_text() if OUT.exists() else ""
        if current != content:
            raise SystemExit("WIRE_PROTOCOL.md is out of date; run scripts/gen_protocol_doc.py")
        return
    OUT.write_text(content)


if __name__ == "__main__":
    main()
